tìm hiểu thuộc tính position
        position: relative; // giá trị tương đối
        position: absolute; // giá trị tuyệt đối 
        position: static; // tĩnh ( giá trị mặc định )

        thuộc tính tọa độ 
        top : bên trên 
        left : bên trái 
        right : bên phải 
        bottom : bên dưới
